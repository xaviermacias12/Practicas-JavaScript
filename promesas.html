Ejercicio 1: Suma de Dos Números con Callback

Este es el primer ejemplo del PDF, que muestra cómo utilizar un callback para manejar el resultado de una operación.

Código JavaScript

// Definimos una función que realiza la suma
function sumar(a, b, callback) {
    const resultado = a + b; // Realiza la suma
    callback(resultado); // Llama al callback con el resultado
}

// Definimos un callback que maneja el resultado
function mostrarResultado(resultado) {
    document.write("La suma es:", resultado);
}

// Llamamos a la función sumar y pasamos los números y el callback
sumar(5, 10, mostrarResultado);

Ejercicio 2: Callback con setTimeout

Este ejemplo muestra cómo usar un callback con una operación asíncrona simulada con setTimeout.

Código JavaScript

// Definimos una función que simula una operación asíncrona
function obtenerDatos(callback) {
    setTimeout(() => {
        const datos = { nombre: "Juan", edad: 30 };
        // Llamamos al callback con los datos obtenidos
        callback(datos);
    }, 2000); // Simulamos un retraso de 2 segundos
}

// Definimos un callback que maneja los datos
function mostrarDatos(datos) {
    document.write('Nombre: ${datos.nombre}, Edad: ${datos.edad}');
}

// Llamamos a la función obtenerDatos y pasamos mostrarDatos como callback
obtenerDatos(mostrarDatos);

Ejercicio 3: Estructura Básica de una Promesa

Las promesas representan la finalización (o falla) de una operación asíncrona y su valor resultante.

Código JavaScript

// Creación de una promesa
const miPromesa = new Promise((resolve, reject) => {
    // Simulamos una operación asíncrona
    setTimeout(() => {
        const exito = true; // Cambiar a false para simular un error
        
        if (exito) {
            resolve("¡Operación completada con éxito!");
        } else {
            reject("¡Algo salió mal!");
        }
    }, 1500);
});

// Uso de la promesa
miPromesa
    .then(resultado => {
        console.log(resultado);
    })
    .catch(error => {
        console.error(error);
    });

Ejercicio 4: Fetch API con Promesas

Uso de fetch con then/catch para obtener datos de una API.

Código JavaScript

const obtenerPosts = () => {
    fetch('https://jsonplaceholder.typicode.com/posts')
        .then(response => response.json())
        .then(data => mostrarPosts(data))
        .catch(error => console.log('Error:', error));
};

// Función para mostrar los posts en el HTML
const mostrarPosts = (posts) => {
    const postsUI = document.getElementById('posts');
    postsUI.innerHTML = ''; // Limpiamos la lista antes de agregar nuevos posts
    posts.forEach(post => {
        const li = document.createElement('li');
        li.textContent = post.title;
        postsUI.appendChild(li);
    });
};

// Llamamos a la función para obtener y mostrar los posts
obtenerPosts();

Ejercicio 5: Async/Await

Uso de async/await para manejar operaciones asíncronas de forma más legible.

Código JavaScript

// Función async que usa await
async function obtenerPostsConAsyncAwait() {
    try {
        let response = await fetch('https://jsonplaceholder.typicode.com/posts');
        if (!response.ok) {
            throw new Error('Error en la llamada a la API');
        }
        let posts = await response.json();
        mostrarPosts(posts);
    } catch (error) {
        console.log('Error:', error);
    }
}

// Función para mostrar los posts
function mostrarPosts(posts) {
    const postsUL = document.getElementById('posts');
    posts.forEach(post => {
        const li = document.createElement('li');
        li.textContent = post.title;
        postsUL.appendChild(li);
    });
}

// Llamada a la función
obtenerPostsConAsyncAwait();

Ejercicio 7: Async/Await para Usuarios

Versión con async/await del ejercicio anterior.

Código JavaScript

// Función async para obtener usuarios
async function obtenerUsuariosAsync() {
    try {
        const response = await fetch('https://jsonplaceholder.typicode.com/users');
        if (!response.ok) {
            throw new Error('Error en la respuesta de la API');
        }
        const users = await response.json();
        mostrarUsuariosAsync(users);
    } catch (error) {
        console.error('Error:', error);
    }
}

// Función para mostrar usuarios
function mostrarUsuariosAsync(users) {
    const usersList = document.getElementById('usersListAsync');
    usersList.innerHTML = '';
    
    users.forEach(user => {
        const userCard = document.createElement('div');
        userCard.className = 'user-card';
        userCard.innerHTML = `
            
${user.name}

            
Email: ${user.email}

            
Teléfono: ${user.phone}

            
Compañía: ${user.company.name}

        `;
        usersList.appendChild(userCard);
    });
}

// Llamar a la función
obtenerUsuariosAsync();
