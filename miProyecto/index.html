Ejercicio Personal: Módulo de Manipulación de Cadenas

Este ejercicio crea un módulo para manipular cadenas de texto.

strings.js

// strings.js
export const mayusculas = (texto) => texto.toUpperCase();
export const minusculas = (texto) => texto.toLowerCase();
export const longitud = (texto) => texto.length;
export const invertir = (texto) => texto.split('').reverse().join('');

main-strings.js

import { mayusculas, minusculas, longitud, invertir } from './strings.js';

const texto = "Hola Mundo";
console.log(mayusculas(texto)); // "HOLA MUNDO"
console.log(minusculas(texto)); // "hola mundo"
console.log(longitud(texto)); // 10
console.log(invertir(texto)); // "odnuM aloH"
